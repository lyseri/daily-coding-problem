Coding Challenge #7

Solution:

        
        function helper(str, n) {
            if (n == 0) {
                return 1;
            }
        
            start = str.length - n;
            if (n[0] == '0') {
                return 0;
            }
        
            result = helper(str, n-1);
            if (n >= 2 && str[start] <= 26) {
                result += helper(str, n-2);
            }
            return result;
        }
        function numWays(str) {
            return helper(str, str.length)
        }
        
        console.log(numWays('1111'));
